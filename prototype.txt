// --- TIME HELPER FUNCTIONS ---
const DAILY_CAP_MINS = 480; // 8 hours

const formatMins = (totalMins) => {
  if (totalMins <= 0) return '0h';
  const h = Math.floor(totalMins / 60);
  const m = totalMins % 60;
  if (h > 0 && m > 0) return `${h}h ${m}m`;
  if (h > 0) return `${h}h`;
  return `${m}m`;
};

const parseMins = (str) => {
  const hMatch = str.match(/(\d+)\s*h/);
  const mMatch = str.match(/(\d+)\s*m/);
  const h = hMatch ? parseInt(hMatch[1]) * 60 : 0;
  const m = mMatch ? parseInt(mMatch[1]) : 0;
  if (!hMatch && !mMatch && !isNaN(parseInt(str))) return parseInt(str) * 60; // default input numbers to hours
  return h + m;
};

                          ? 'bg-[#f0f4ff] text-[#172b4d] border-blue-100'
                          : 'bg-[#fffbeb] text-[#92400e] border-[#fde68a]')
                      : (isOverLimit ? 'bg-[#fffbeb] text-[#92400e] border-[#fde68a]' : 'hidden')
                  }`}>
                    <div className="flex items-start gap-2">
                      {!isEditedByUser
                        ? (remainingIsPositive && !actualExceedsRemaining
                            ? <Info size={16} className="text-blue-500 mt-0.5 shrink-0" />
                            : <div className="relative group shrink-0">
                                <AlertTriangle size={16} className="text-[#d97706] mt-0.5" />
                                {!remainingIsPositive && (
                                  <div className="absolute bottom-full left-1/2 -translate-x-1/2 mb-2 hidden group-hover:block bg-gray-800 text-white text-[11px] rounded px-2 py-1 whitespace-nowrap shadow-lg z-50">
                                    You are in the Extra/OT zone. Please activate the Extra/OT button to record your overtime.
                                    <div className="absolute top-full left-1/2 -translate-x-1/2 w-0 h-0 border-l-4 border-r-4 border-t-4 border-l-transparent border-r-transparent border-t-gray-800"></div>
                                  </div>
                                )}
                              </div>)
                        : (isOverLimit ? <AlertTriangle size={16} className="text-[#d97706] mt-0.5 shrink-0" /> : null)
                      }

                      <div>
                        {/* End Timer — Not edited, remaining > 0, actual ≤ remaining */}
                        {!isEditedByUser && remainingIsPositive && !actualExceedsRemaining && (
                          <span className="font-medium">
                            Actual logged time: {formatMins(actualMins)}. Remaining today: {formatMins(remainingTotal)}.
                          </span>
                        )}

                        {/* End Timer — Not edited, remaining > 0, actual > remaining */}
                        {!isEditedByUser && remainingIsPositive && actualExceedsRemaining && (
                          <span className="font-bold">
                            Actual logged time: {formatMins(actualMins)}. You are Extra/OT by {formatMins(effectiveOverMins)}.
                          </span>
                        )}

                        {/* End Timer — Not edited, remaining ≤ 0 */}
                        {!isEditedByUser && !remainingIsPositive && (
                          <span className="font-bold block">
                            You have reached today's 8-hour standard threshold. You are Extra/OT by {formatMins(effectiveOverMins)}.
                          </span>
                        )}

                        {/* End Timer — User edited and over limit */}
                        {isEditedByUser && isOverLimit && (
                          <span className="font-bold">
                            You are Extra/OT by {formatMins(effectiveOverMins)} compared to today's standard hours.
                          </span>
                        )}
                      </div>
                    </div>
                  </div>
                )}

                {/* ---- MANUAL LOG FLOW (FR-05) ---- */}
                {!isTimerMode && (
                  <div className={`p-3 rounded text-[13px] border ${
                    (!isEditedByUser && remainingTotal > 0) || (isEditedByUser && !isOverLimit)
                      ? 'bg-[#f0f4ff] text-[#172b4d] border-blue-100'
                      : 'bg-[#fffbeb] text-[#92400e] border-[#fde68a]'
                  }`}>
                    <div className="flex items-start gap-2">
                      {((!isEditedByUser && remainingTotal > 0) || (isEditedByUser && !isOverLimit))
                        ? <Info size={16} className="text-blue-500 mt-0.5 shrink-0" />
                        : <div className="relative group shrink-0">
                            <AlertTriangle size={16} className="text-[#d97706] mt-0.5" />
                            {!remainingIsPositive && (
                              <div className="absolute bottom-full left-1/2 -translate-x-1/2 mb-2 hidden group-hover:block bg-gray-800 text-white text-[11px] rounded px-2 py-1 whitespace-nowrap shadow-lg z-50">
                                You are in the Extra/OT zone. Please activate the Extra/OT button to record your overtime.
                                <div className="absolute top-full left-1/2 -translate-x-1/2 w-0 h-0 border-l-4 border-r-4 border-t-4 border-l-transparent border-r-transparent border-t-gray-800"></div>
                              </div>
                            )}
                          </div>
                      }

                      <div>
                        {/* Manual: Not over limit */}
                        {(!isEditedByUser && remainingTotal > 0) && (
                          <span className="font-medium">You have {formatMins(remainingTotal)} of standard time left to log today.</span>
                        )}

                        {/* Manual: Just opened, already over limit (remaining ≤ 0) */}
                        {!isEditedByUser && remainingTotal <= 0 && (
                          <span className="font-bold block">
                            You have reached today's 8-hour standard threshold. You are Extra/OT by {formatMins(effectiveOverMins)}.
                          </span>
                        )}

                        {/* Manual: Edited to be over limit */}
                        {isEditedByUser && isOverLimit && (
                          <span className="font-bold">
                            If saved, today's total will be {formatMins(projectedTotal)} — you are Extra/OT by {formatMins(effectiveOverMins)} compared to standard hours.
                          </span>
                        )}
                      </div>
                    </div>
                  </div>
                )}

                {/* Description Section */}
                <div>
                  <textarea placeholder="Description" className="w-full border border-gray-300 rounded px-3 py-2 text-[13px] h-20 resize-none focus:outline-none focus:border-blue-500"></textarea>
                </div>
              </div>
            </div>

            {/* Modal Footer */}
            <div className="px-6 py-4 border-t border-gray-200 flex justify-between items-center rounded-b">
              {/* Daily Total Indicator (PDF Spec) */}
              <div className="flex items-center bg-blue-50 text-blue-700 px-3 py-1.5 rounded-full border border-blue-100 shadow-sm">
                 <div className="w-2 h-2 rounded-full bg-blue-600 mr-2"></div>
                 <span className="text-xs font-semibold">{formatMins(totalLoggedMins)} logged today</span>
              </div>

              <div className="flex gap-3">
                <button onClick={() => setIsLogWorkModalOpen(false)} className="text-[#42526e] border border-gray-300 px-4 py-1.5 rounded text-[13px] font-bold hover:bg-gray-50">
                  Discard
                </button>
                <button onClick={() => setIsLogWorkModalOpen(false)} className="bg-[#2563eb] text-white px-6 py-1.5 rounded text-[13px] font-bold hover:bg-blue-700 shadow-sm">
                  Submit
                </button>
              </div>
            </div>
          </div>
        </div>
      )}

      {/* Global Styles */}
      <style dangerouslySetInnerHTML={{__html: `
        .custom-scrollbar::-webkit-scrollbar { width: 6px; }
        .custom-scrollbar::-webkit-scrollbar-track { background: transparent; }
        .custom-scrollbar::-webkit-scrollbar-thumb { background-color: #cbd5e1; border-radius: 10px; }
      `}} />
    </div>
  );
}
